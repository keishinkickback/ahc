torch.setdefaulttensortype('torch.FloatTensor')
agg = require 'aggclustering'
Plot = require 'itorch.Plot'

function plotClusters(x, groups, nClusters)
    plot = Plot()
    colortable ={'red', 'blue', 'green', 'yellow', 'black','purple'}
    for ci, g in ipairs(groups) do
        for _, p in ipairs(g) do
            colorIndex = ci % 6 + 1
            plot:circle(x[{{p},{1}}]:reshape(1), x[{{p},{2}}]:reshape(1), colortable[colorIndex])
            :text(x[{{p},{1}}]:reshape(1), x[{{p},{2}}]:reshape(1), tostring(p), torch.Tensor({0}), 'black')
        end
    end
    plot:title('Scatter Plot: ' ..tostring(nClusters) .. ' clusters')
    plot:xaxis('x_1'):yaxis('x_2')
    plot:draw()
end
